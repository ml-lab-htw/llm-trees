def predict(X: dict):
    nodes = [None] * 2
    nodes[0] = int(X["Round"] <= 6)
    nodes[1] = int(X["Official"] == 1)
    
    if nodes[0]:
        prediction = 1  # de la Hoya
    else:
        if nodes[1]:
            prediction = 0  # Trinidad
        else:
            prediction = 1  # de la Hoya
    
    return prediction, nodes
