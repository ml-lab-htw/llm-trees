    nodes[6] = X["pulse"] > 60  # Pulse > 60 beats per minute

    if nodes[0]:
        prediction = 1
    elif nodes[1]:
        if nodes[2]:
            prediction = 1
        elif nodes[3]:
            prediction = 1
        else:
            if nodes[4]:
                prediction = 1
            else:
                prediction = 0
    elif nodes[5]:
        if nodes[6]:
            if nodes[7]:
                prediction = 1
            else:
                prediction = 0
        else:

    else:

    return prediction, nodes
