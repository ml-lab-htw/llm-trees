def predict(X: dict):
    nodes = 2 * [None]
    nodes[0] = X['Type_school'] == 0  # Type_school == primary_terminal_leaver
    nodes[1] = X['DVRT'] <= 50        # DVRT score threshold

    if nodes[0]:
        prediction = 0  # Not taken
    else:
        if nodes[1]:
            prediction = 0  # Not taken
        else:
            prediction = 1  # Taken

    return prediction, nodes
